defined_text = {
	name = SwissMercsLimitReached
	random = no
	text = {
		localisation_key = swiss_mercs_fl_limit_reached
		trigger = {
			# Tech 23-32
			if = {
				limit = {
					mil_tech = 23
				}
				if = {
					limit = {
						land_forcelimit = 1
						NOT = { land_forcelimit = 24 }
					}
					NOT = { sea_swiss_mercenaries_3 = 10 }
				}
				if = {
					limit = {
						land_forcelimit = 24
						NOT = { land_forcelimit = 48 }
					}
					NOT = { sea_swiss_mercenaries_3 = 20 }
				}
				if = {
					limit = {
						land_forcelimit = 48
						NOT = { land_forcelimit = 96 }
					}
					NOT = { sea_swiss_mercenaries_3 = 30 }
				}
				if = {
					limit = {
						land_forcelimit = 96
						NOT = { land_forcelimit = 192 }
					}
					NOT = { sea_swiss_mercenaries_3 = 40 }
				}
				if = {
					limit = {
						land_forcelimit = 192
						NOT = { land_forcelimit = 384 }
					}
					NOT = { sea_swiss_mercenaries_3 = 80 }
				}
				if = {
					limit = {
						land_forcelimit = 384
					}
					NOT = { sea_swiss_mercenaries_3 = 120 }
				}
			}
			# Tech 12-22
			else_if = {
				limit = {
					mil_tech = 12
					NOT = { mil_tech = 23 }
				}
				if = {
					limit = {
						land_forcelimit = 1
						NOT = { land_forcelimit = 24 }
					}
					NOT = { sea_swiss_mercenaries_2 = 10 }
				}
				if = {
					limit = {
						land_forcelimit = 24
						NOT = { land_forcelimit = 48 }
					}
					NOT = { sea_swiss_mercenaries_2 = 20 }
				}
				if = {
					limit = {
						land_forcelimit = 48
						NOT = { land_forcelimit = 96 }
					}
					NOT = { sea_swiss_mercenaries_2 = 30 }
				}
				if = {
					limit = {
						land_forcelimit = 96
						NOT = { land_forcelimit = 192 }
					}
					NOT = { sea_swiss_mercenaries_2 = 40 }
				}
				if = {
					limit = {
						land_forcelimit = 192
					}
					NOT = { sea_swiss_mercenaries_2 = 60 }
				}
			}
			# tech 1-12
			else = {
				if = {
					limit = {
						land_forcelimit = 1
						NOT = { land_forcelimit = 40 }
					}
					NOT = { sea_swiss_mercenaries_1 = 10 }
				}
				if = {
					limit = {
						land_forcelimit = 40
					}
					NOT = { sea_swiss_mercenaries_1 = 20 }
				}
			}
		}
	}
	text = { localisation_key = swiss_mercs_fl_limit_not_reached }
