			limit = {
				NRM = { is_subject = no }

			}
			hidden_effect = {
				release = NRM
			}
			vassalize = NRM
			hidden_effect = {
				NRM = {
					change_government = monarchy
					add_government_reform = feudalism_reform
				}
			}
		}
		NRM = {
			define_heir = {
				name = "Rollo"
				dynasty = "Ragnvaldsson"
				religion = norse_pagan_reformed
				claim = 100
				birth_date = 860.1.1
			}
			hidden_effect = {
				kill_ruler = yes
			}
			add_ruler_modifier = {
				name = live_long_and_prosper_modifier
				duration = -1
			}
		}
		if = {
			limit = {
				normandy_area = {
					is_core = ROOT
				}
			}
			normandy_area = { remove_core = ROOT }
		}
	}
	option = {
		name = sea_denmark_events.1.b
		ai_chance = { factor = 0 }
		add_stability = -1
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = -20
		}
	}
}

			}

		}

			limit = {
				crown_land_share = 25
				NOT = { crown_land_share = 27 }
			}
			take_estate_land_share_massive = { estate = all }
		}
		if = {
			limit = {
				crown_land_share = 24
				NOT = { crown_land_share = 25 }
			}
			take_estate_land_share_massive = { estate = all }
			take_estate_land_share_medium = { estate = all }
