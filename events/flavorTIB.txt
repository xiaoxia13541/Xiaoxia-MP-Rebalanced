# The Dalai Lama takes control
country_event = {
	id = flavor_tib.1
	title = "flavor_tib.EVTNAME1"
	desc = "flavor_tib.EVTDESC1"
	picture = RELIGION_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		is_revolutionary = no
		OR = {
			AND = {
				tag = TIB
				OR = {
					NOT = { dip = 3 }
					NOT = { mil = 3 }
					NOT = { adm = 3 }
				}
				any_owned_province = {
					religion = vajrayana
				}
				NOT = { is_year = 1642 }
				NOT = { government = theocracy }
				NOT = { has_reform = celestial_empire }
			}
			has_country_flag = dalai_lama_flag
		}
	}
	
	mean_time_to_happen = {
		months = 200
	}
	
	option = {
		trigger = {
			NOT = { has_reform = celestial_empire }
		}				
		name = "flavor_tib.EVTOPTA1"
		ai_chance = { factor = 100 }
		if = {
			limit = {
				NOT = {
					religion = vajrayana
				}
			}
			change_religion = vajrayana
		}
		if = {
			limit = {
				NOT = {
					government = theocracy
				}
			}
			change_government = theocracy
			add_government_reform = tibetan_dual_government_reform
			#set_legacy_government = theocratic_government
		}
		else = {
			add_devotion = 20
		}
	}
	option = {
		name = "flavor_tib.EVTOPTB1"
		ai_chance = { factor = 0 }
		add_stability_or_adm_power = yes
	}
}

# Tibetan relations with the Hordes by Daniel Tolman
country_event = {
	id = flavor_tib.2
	title = "flavor_tib.EVTNAME2"
	desc = "flavor_tib.EVTDESC2"
	picture = HORDE_ON_HORSEBACK_eventPicture
	
	fire_only_once = yes
	
	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = tibetan_government_flag # trigger for tibet missions if tag is Khoshuud, Dzungar, or Khalsa
		}
	}
	
	option = { # Improve relations with the hordes
		name = "flavor_tib.EVTOPTA2"
		ai_chance = { factor = 100 }
		every_country = {
			limit = {
				OR = {
					primary_culture = altaic_new 
					culture_group = altaic
				}
				has_reform = steppe_horde
			}
			add_opinion = {
				who = ROOT
				modifier = tibetan_steppe_relations
			}
		}
	}
	option = { # Khalkha Mongols
		name = "flavor_tib.EVTOPTB2"
		ai_chance = { factor = 0 }
		trigger = {
			NOT = {
				exists = KLK
			}
			NOT = { has_reform = celestial_empire }
			was_never_end_game_tag_trigger = yes
		}
		add_stability = -2
		change_tag = KLK
		on_change_tag_effect = yes
		change_government = tribal
		add_government_reform = steppe_horde
		change_primary_culture = khalkha
		if = {
			limit = {
				has_updated_culture = { culture = tibetan }
			}
			add_accepted_culture = tibetan_new
		}
		else = {
			add_accepted_culture = tibetan
		}
		if = {
			limit = {
				is_year = 1527
				NOT = { is_year = 1582 }
			}
			define_ruler = {
				name = "Altan"
				dynasty = "Borjigin"
				culture = khalkha
				adm = 4
				dip = 4
				mil = 5
			}
		}
		else = {
			define_ruler = {
				dynasty = "Borjigin"
				culture = khalkha
			}
		}
	}
	option = { # Khoshuud Oirats
		name = "flavor_tib.EVTOPTC2"
		ai_chance = { factor = 0 }
		trigger = {
			NOT = {
				exists = KSD
			}
			NOT = { has_reform = celestial_empire }
			was_never_end_game_tag_trigger = yes
		}
		add_stability = -2
		change_tag = KSD
		on_change_tag_effect = yes
		change_primary_culture = oirats
		if = {
			limit = {
				has_updated_culture = { culture = tibetan }
			}
			add_accepted_culture = tibetan_new
		}
		else = {
			add_accepted_culture = tibetan
		}
		change_government = tribal
		add_government_reform = steppe_horde
		if = {
			limit = {
				is_year = 1600
				NOT = { is_year = 1655 }
			}
			define_ruler = {
				name = "Gushri"
				dynasty = "Khoshut"
				culture = oirats
				adm = 2
				dip = 4
				mil = 5
			}
		}
		else = {
			define_ruler = {
				dynasty = "Khoshut"
				culture = oirats
			}
		}
	}
	option = { # Dzungar Oirats
		name = "flavor_tib.EVTOPTD2"
		ai_chance = { factor = 0 }
		trigger = {
			is_year = 1634
			NOT = {
				exists = ZUN
			}
			NOT = { has_reform = celestial_empire }
			was_never_end_game_tag_trigger = yes
		}
		add_stability = -2
		change_tag = ZUN
		on_change_tag_effect = yes
		change_primary_culture = oirats
		if = {
			limit = {
				has_updated_culture = { culture = tibetan }
			}
			add_accepted_culture = tibetan_new
		}
		else = {
			add_accepted_culture = tibetan
		}
		change_government = tribal
		add_government_reform = steppe_horde
		define_ruler = {
			dynasty = "Choros"
			culture = oirats
		}
	}
}
